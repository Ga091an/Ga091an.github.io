      # Our decoder will embed the inputs before feeding them to the LSTM.
  self.embed_tokens = nn.Embedding(
      num_embeddings=len(dictionary),
      embedding_dim=embed_dim,
      padding_idx=dictionary.pad(),
  )
  self.dropout = nn.Dropout(p=dropout)

  # We'll use a single-layer, unidirectional LSTM for simplicity.
  self.lstm = nn.LSTM(
      # For the first layer we'll concatenate the Encoder's final hidden
      # state with the embedded target tokens.
      input_size= embed_dim,
      hidden_size=hidden_dim,
      num_layers=1,
      bidirectional=False,
  )

  # Define the output projection.
  self.x_projection=nn.Linear(2*hidden_dim,hidden_dim)
  self.ot_projection=nn.Linear(2*hidden_dim, hidden_dim)
  self.output_projection = nn.Linear(hidden_dim, len(dictionary))

  """
  if incremental_state is not None:

class Solution { public: vector singleNumbers(vector& nums) { int n=nums.size(); if(n==0) return vector(); int r=0; //两个相同的数字做异或(^)运算，结果是0 //因此对数组中的所有数字做两两做异或运算，最后会剩下两个不同的数字（也就是想要的结果） //把这两个数字写成二进制的形式，那么在某一位，这两个数字肯定是不一样的，否则就是相同的数字了 //异或运算中，相同取0，不同取1，那么最后的结果中，从右往左第一个为1的位就是这两个不同的数字不相同的第一位 //例如 2 2 4 4 3 5 作异或，等于 3 5 异或 // 3 = 0011 // 5 = 0101 // r=3^5= 0110 // 从右往左r的第二位为1，这就是3和5第一次不同的位（从右往左看） for(auto s:nums) r^=s; int mask=1; //mask负责找出这个位的位置 while(!(r&mask)){; mask<<=1; } //依据这个位是1还是0，把数字分组，不同的两个数一定是分到不同的组的 int e1=0,e2=0; for(auto s:nums){ if(s&mask){ e1=s; } else e2=s; } vector res(2); res[0]=e1;res[1]=e2; return res; } };
