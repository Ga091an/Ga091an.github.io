lc-Gray code

Posted on 2019-09-21

以[1,2,2]为例：[]->[],1->[],1,2,12->[],1,2,12,2,12,22,122。 所以我们用last_insert来记录上一个处理的数字的个数，然后判定当前的数字和上面的是否相同，若不同，则循环还是从0到当前子集的个数，若相同， 则从size-last_insert的位置开始插入，这样就不会产生重复了。

class Solution {
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        int n=nums.size();
        vector<vector<int>> res;
        res.push_back(vector<int>{});
        sort(nums.begin(),nums.end());
        int ct,last_insert=1;
        for(int i=0;i<n;i++){
            int size=res.size();
            ct=last_insert;
            last_insert=0;
            if(i>0&&nums[i]==nums[i-1]){
                while(ct){
                    last_insert++;
                    res.push_back(res[size-ct]);
                    res.back().push_back(nums[i]);
                    ct--;
                }
            }
            else{
                for(int j=0;j<size;j++){
                    last_insert++;
                    res.push_back(res[j]);
                    res.back().push_back(nums[i]);
                }   
            }
        }
        return res;
    }
};
